```julia
using Plots
```

```julia
x = range(-pi,pi,length=100)
y = sin.(x)
plot(aspect_ratio=:equal,xlim=[-pi,pi])
plot!(sin,color=:red)
plot!(x, y, fill=y .+ 1,color=:green)
plot!(x, y, fill=y .- 1,color=:blue)
```

```julia
x = range(-pi,pi,length=100)
y = sin.(x)
l = -ones(length(y))
u = ones(length(y))
plot(aspect_ratio=:equal,xlim=[-pi,pi])
plot!(sin, ribbons=[l, u], color=:blue,label=:none)
plot!(sin,color=:red,label=:none)
```
